      document.addEventListener('DOMContentLoaded', function() {
  // Get all navigation links
  const navLinks = document.querySelectorAll('.nav-link');
  
  // Add click event listeners to each nav link
  navLinks.forEach(link => {
    link.addEventListener('click', function(e) {
      // Remove active class from all links
      navLinks.forEach(l => l.classList.remove('text-green-500'));
      
      // Add active class to clicked link
      this.classList.add('text-green-500');
      
      // Store active link in localStorage
      localStorage.setItem('activeNavLink', this.dataset.target);
    });
  });
  
  // Get all links on the page (not just nav links)
  const allLinks = document.querySelectorAll('a:not(.nav-link)');
  
  // Add click event to all other links to reset active state
  allLinks.forEach(link => {
    link.addEventListener('click', function(e) {
      // Remove active class from all nav links
      navLinks.forEach(navLink => navLink.classList.remove('text-green-500'));
      
      // Also clear localStorage
      localStorage.removeItem('activeNavLink');
    });
  });
  
  // On page load, check if we should set an active link
  const currentPath = window.location.pathname;
  let activeSet = false;
  
  // Try to match based on the current path
  navLinks.forEach(link => {
    const href = link.getAttribute('href');
    const hrefFile = href.split('/').pop();
    const pathFile = currentPath.split('/').pop();
    
    // If current page matches this nav link
    if (pathFile === hrefFile) {
      navLinks.forEach(l => l.classList.remove('text-green-500'));
      link.classList.add('text-green-500');
      localStorage.setItem('activeNavLink', link.dataset.target);
      activeSet = true;
    }
  });
  
  // If no match by URL but we have localStorage, use that
  if (!activeSet) {
    const activeLink = localStorage.getItem('activeNavLink');
    if (activeLink) {
      const link = document.querySelector(`[data-target="${activeLink}"]`);
      if (link) {
        navLinks.forEach(l => l.classList.remove('text-green-500'));
        link.classList.add('text-green-500');
      }
    }
  }
});

<script>
  document.addEventListener('DOMContentLoaded', function() {
    // Get sign-in buttons by their IDs
    const desktopSignInBtn = document.getElementById('desktop-sign-in');
    const mobileSignInBtn = document.getElementById('mobile-sign-in');
    const sidebar = document.getElementById('sign-in-sidebar');
    const closeBtn = document.getElementById('close-sidebar');
    const overlay = document.querySelector('.sidebar-overlay');
    const phoneScreen = document.getElementById('phone-screen');
    const otpScreen = document.getElementById('otp-screen');
    const continueBtn = document.getElementById('continue-btn');
    const phoneInput = document.getElementById('phone-input');
    const displayNumber = document.getElementById('display-number');
    const submitBtn = document.getElementById('submit-btn');
    const changeNumberBtn = document.getElementById('change-number');
    const otpInputs = document.querySelectorAll('.otp-input');
    
    // Function to open sidebar
    function openSidebar() {
      if (sidebar) {
        sidebar.classList.add('active');
      }
    }
    
    // Function to close sidebar
    function closeSidebar() {
      if (sidebar) {
        sidebar.classList.remove('active');
      }
    }
    
    // Add click event listeners to sign-in buttons
    if (desktopSignInBtn) {
      desktopSignInBtn.addEventListener('click', function(e) {
        e.preventDefault();
        openSidebar();
      });
    }
    
    if (mobileSignInBtn) {
      mobileSignInBtn.addEventListener('click', function(e) {
        e.preventDefault();
        openSidebar();
      });
    }
    
    // Close button functionality
    if (closeBtn) {
      closeBtn.addEventListener('click', function() {
        closeSidebar();
        
        // Reset to phone screen when closing
        if (phoneScreen && otpScreen) {
          phoneScreen.classList.remove('hidden');
          otpScreen.classList.add('hidden');
        }
      });
    }
    
    // Close when clicking overlay
    if (overlay) {
      overlay.addEventListener('click', function() {
        closeSidebar();
        
        // Reset to phone screen when closing
        if (phoneScreen && otpScreen) {
          phoneScreen.classList.remove('hidden');
          otpScreen.classList.add('hidden');
        }
      });
    }
    
    // Phone input validation to enable continue button
    if (phoneInput && continueBtn) {
      phoneInput.addEventListener('input', function() {
        if (this.value.length === 11) {
          continueBtn.classList.remove('bg-gray-100', 'text-gray-400', 'disabled', 'cursor-not-allowed');
          continueBtn.classList.add('bg-green-500', 'text-white', 'hover:bg-green-600');
          continueBtn.disabled = false;
        } else {
          continueBtn.classList.add('bg-gray-100', 'text-gray-400', 'disabled', 'cursor-not-allowed');
          continueBtn.classList.remove('bg-green-500', 'text-white', 'hover:bg-green-600');
          continueBtn.disabled = true;
        }
      });
    }
    
    // Continue button to show OTP screen
    if (continueBtn && phoneScreen && otpScreen && displayNumber && phoneInput) {
      continueBtn.addEventListener('click', function() {
        if (phoneInput.value.length === 11) {
          // Show the phone number in the OTP screen
          displayNumber.textContent = '+88' + phoneInput.value;
          
          // Show OTP screen, hide phone screen
          phoneScreen.classList.add('hidden');
          otpScreen.classList.remove('hidden');
          
          // Start the timer
          startTimer();
          
          // Focus the first OTP input
          if (otpInputs[0]) {
            otpInputs[0].focus();
          }
        }
      });
    }
    
    // Change number button functionality
    if (changeNumberBtn && phoneScreen && otpScreen) {
      changeNumberBtn.addEventListener('click', function(e) {
        e.preventDefault();
        
        // Show phone screen, hide OTP screen
        phoneScreen.classList.remove('hidden');
        otpScreen.classList.add('hidden');
        
        // Focus on phone input
        if (phoneInput) {
          phoneInput.focus();
        }
      });
    }
    
    // OTP input functionality
    if (otpInputs.length) {
      otpInputs.forEach((input, index) => {
        // Move to the next input when a character is entered
        input.addEventListener('input', function() {
          if (this.value.length === 1) {
            if (index < otpInputs.length - 1) {
              otpInputs[index + 1].focus();
            } else {
              // Check if all inputs are filled
              checkOtpCompletion();
            }
          }
        });
        
        // Handle backspace to go to previous input
        input.addEventListener('keydown', function(e) {
          if (e.key === 'Backspace' && !this.value && index > 0) {
            otpInputs[index - 1].focus();
          }
        });
        
        // Allow only numbers
        input.addEventListener('keypress', function(e) {
          if (isNaN(parseInt(e.key))) {
            e.preventDefault();
          }
        });
      });
    }
    
    // Function to check if all OTP inputs are filled
    function checkOtpCompletion() {
      let isComplete = true;
      
      otpInputs.forEach(input => {
        if (!input.value) {
          isComplete = false;
        }
      });
      
      if (isComplete && submitBtn) {
        submitBtn.classList.remove('bg-gray-100', 'text-gray-400', 'disabled', 'cursor-not-allowed');
        submitBtn.classList.add('bg-green-500', 'text-white', 'hover:bg-green-600');
        submitBtn.disabled = false;
      }
    }
    
    // Timer functionality
    function startTimer() {
      const timerElement = document.getElementById('timer');
      if (!timerElement) return;
      
      let timeLeft = 119; // 1:59 in seconds
      
      // Update timer text
      function updateTimer() {
        const minutes = Math.floor(timeLeft / 60);
        const seconds = timeLeft % 60;
        
        timerElement.textContent = `${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
        
        if (timeLeft === 0) {
          clearInterval(interval);
          
          // Add resend code functionality here
          const resendLink = document.createElement('a');
          resendLink.textContent = 'Resend Code';
          resendLink.href = '#';
          resendLink.className = 'text-green-500 font-medium ml-2';
          resendLink.addEventListener('click', function(e) {
            e.preventDefault();
            startTimer();
            this.remove();
          });
          
          timerElement.textContent = '';
          timerElement.parentNode.appendChild(resendLink);
        }
        
        timeLeft--;
      }
      
      // Update immediately and then every second
      updateTimer();
      const interval = setInterval(updateTimer, 1000);
    }
    
    // Submit button functionality
    if (submitBtn) {
      submitBtn.addEventListener('click', function() {
        // Handle verification here
        // For demo, we'll just close the sidebar
        alert('Verification successful!'); // Replace with your actual verification logic
        closeSidebar();
        
        // Reset to phone screen for next time
        if (phoneScreen && otpScreen) {
          phoneScreen.classList.remove('hidden');
          otpScreen.classList.add('hidden');
        }
        
        // Reset inputs
        if (phoneInput) phoneInput.value = '';
        otpInputs.forEach(input => input.value = '');
        
        // Reset buttons
        if (continueBtn) {
          continueBtn.classList.add('bg-gray-100', 'text-gray-400', 'disabled', 'cursor-not-allowed');
          continueBtn.classList.remove('bg-green-500', 'text-white', 'hover:bg-green-600');
          continueBtn.disabled = true;
        }
        
        submitBtn.classList.add('bg-gray-100', 'text-gray-400', 'disabled', 'cursor-not-allowed');
        submitBtn.classList.remove('bg-green-500', 'text-white', 'hover:bg-green-600');
        submitBtn.disabled = true;
      });
    }
  });
